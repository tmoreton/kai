kai.config.ts
// Ask Kai to scaffold a new API endpoint
import { Kai } from 'kai';

const agent = new Kai({
  model: 'claude-3-opus',
  context: './src',
  tools: ['git', 'codegen', 'deploy']
});

// "Create a REST endpoint for user authentication"
await agent.generate({
  prompt: 'Create a secure login endpoint with JWT tokens',
  output: './src/auth/login.ts'
});
